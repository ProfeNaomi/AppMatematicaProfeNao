function addStep(title, formula, detail) { const container = document.getElementById('steps-container'); const step = document.createElement('div'); step.className = 'step-card'; step.innerHTML = `
${title} ${detail}
${formula}
`; container.appendChild(step); } function processEquation() { let input = document.getElementById('equation-input').value.toLowerCase(); const container = document.getElementById('steps-container'); const status = document.getElementById('status-tag'); container.innerHTML = ""; // Validación de igualdad if (!input.includes('=')) { container.innerHTML = "
❌ ERROR: No detecto el signo '='.
"; return; } // --- MOTOR DE AUTOCORRECCIÓN --- input = input.replace(/raiz/g, 'sqrt'); input = input.replace(/(\d)([a-z])/g, '$1*$2'); input = input.replace(/(\d)(\()/g, '$1*$2'); input = input.replace(/(\))(\d)/g, '$1*$2'); try { let parts = input.split('='); let leftSide = parts[0]; let rightSide = parts[1]; addStep("Escaneo del Sistema", `${leftSide} = ${rightSide}`, "Entrada normalizada"); // Simplificación con Math.js let simplifiedLeft = math.simplify(leftSide); let simplifiedRight = math.simplify(rightSide); addStep("Simplificación", `${simplifiedLeft.toString()} = ${simplifiedRight.toString()}`, "Resolviendo raíces y constantes"); // Convertir a forma: expresión = 0 const equation = math.parse(`(${simplifiedLeft.toString()}) - (${simplifiedRight.toString()})`); // Obtener coeficientes const a = math.derivative(equation, 'x').evaluate({x: 1}); const b = equation.evaluate({x: 0}); if (a === 0) { if (b === 0) { addStep("Análisis Final", "0 = 0", "INFINITAS SOLUCIONES (Identidad)"); } else { addStep("Análisis Final", `${b} = 0`, "SIN SOLUCIÓN (Contradicción)"); } } else { let xValue = -b / a; addStep("Trasposición", `${a}x = ${-b}`, "Agrupando términos"); addStep("Despeje", `x = ${-b} / ${a}`, "Aislando x"); const finalResult = math.format(xValue, {precision: 4}); addStep("SOLUCIÓN", `x = ${finalResult}`, "Análisis completado"); } status.innerText = "ANÁLISIS COMPLETADO"; status.className = "text-green-400 italic text-[10px]"; } catch (err) { container.innerHTML = `
ERROR: Verifica que la ecuación sea de grado 1.
`; } }
