function addStep(title, formula, detail) { const container = document.getElementById('steps-container'); const step = document.createElement('div'); step.className = 'step-card'; step.innerHTML = `
${title} ${detail}
${formula}
`; container.appendChild(step); } function processEquation() { let input = document.getElementById('equation-input').value.toLowerCase(); const container = document.getElementById('steps-container'); const status = document.getElementById('status-tag'); container.innerHTML = ''; if (!input.includes('=')) { container.innerHTML = "
❌ ERROR: No detecto el signo '='. Una ecuación requiere una igualdad.
"; return; } // --- MOTOR DE AUTOCORRECCIÓN (Para evitar errores de sintaxis) --- // 1. Reemplaza 'raiz' por 'sqrt' input = input.replace(/raiz/g, 'sqrt'); // 2. Inserta '*' entre números y letras (ej: 2x -> 2*x) input = input.replace(/(\d)([a-z])/g, '$1*$2'); // 3. Inserta '*' entre paréntesis y números/letras (ej: 2(x) -> 2*(x)) input = input.replace(/(\d)(\()/g, '$1*$2'); input = input.replace(/(\))(\d)/g, '$1*$2'); try { let [leftSide, rightSide] = input.split('='); addStep("Escaneo del Sistema", `${leftSide} = ${rightSide}`, "Entrada normalizada"); // Simplificar ambos lados (resuelve raíces y suma términos semejantes) let simplifiedLeft = math.simplify(leftSide); let simplifiedRight = math.simplify(rightSide); addStep("Simplificación", `${simplifiedLeft.toString()} = ${simplifiedRight.toString()}`, "Resolviendo raíces y constantes"); // Creamos la expresión: izquierda - derecha = 0 const equation = math.parse(`(${simplifiedLeft.toString()}) - (${simplifiedRight.toString()})`); // Extraemos coeficientes de x y la constante // Derivada de ax + b respecto a x es 'a' const a = math.derivative(equation, 'x').evaluate({x: 1}); // Si evaluamos x=0, obtenemos la constante 'b' const b = equation.evaluate({x: 0}); // Análisis de Casos Especiales if (a === 0) { if (b === 0) { addStep("Análisis Final", "0 = 0", "INFINITAS SOLUCIONES: Es una identidad matemática."); } else { addStep("Análisis Final", `${b} = 0`, "SIN SOLUCIÓN: La igualdad es falsa (Contradicción)."); } } else { // Despeje: ax + b = 0 => x = -b / a let xValue = -b / a; addStep("Trasposición", `${a}x = ${-b}`, "Agrupando términos con x"); addStep("Despeje", `x = ${-b} / ${a}`, "Aislando la variable"); // Resultado final limpio const finalResult = math.format(xValue, {precision: 4}); addStep("SOLUCIÓN", `x = ${finalResult}`, "Proceso de despeje finalizado con éxito."); } status.innerText = "ANÁLISIS COMPLETADO"; status.className = "text-green-400 italic text-[10px]"; } catch (err) { container.innerHTML = `
ERROR DE PROCESAMIENTO:
Asegúrate de que la ecuación sea lineal (grado 1).
Detalle técnico: ${err.message}
`; } }
